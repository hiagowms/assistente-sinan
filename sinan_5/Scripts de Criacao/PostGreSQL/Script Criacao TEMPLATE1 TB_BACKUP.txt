START WITH 1
  MINVALUE 1
  CACHE 1;

   TP_BANCO           VARCHAR(2)                     NOT NULL

      CONSTRAINT NN_BACKUP_SGSISTEMA CHECK (SG_SISTEMA_BACKUP IS NOT NULL),
   CONSTRAINT PK_BACKUP PRIMARY KEY (CO_SEQ_BACKUP)
