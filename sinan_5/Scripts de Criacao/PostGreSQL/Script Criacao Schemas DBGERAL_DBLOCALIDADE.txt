-- SCRIPT RESPONSAVEL PELA CRIAÇÃO DOS SCHEMAS NO BANCO DE DADOS POSTGRESQL

CREATE SCHEMA dbcspu AUTHORIZATION cspuweb;
COMMENT ON SCHEMA dbcspu IS 'Este schema contém tabelas do sistema CSPU';

CREATE SCHEMA dbgeral AUTHORIZATION geral;
COMMENT ON SCHEMA dbgeral IS 'Schema do DBGeral. Relação de tabelas padrão para todos os sistemas';

CREATE SCHEMA dblocalidade AUTHORIZATION localidadeweb;
COMMENT ON SCHEMA dblocalidade IS 'Schema do DBLocalidade. Relação de tabelas padrão para todos os sistemas, inclusive para o schema DBGeral';

CREATE SCHEMA dbsisnet AUTHORIZATION sisnetweb;
COMMENT ON SCHEMA dbsisnet IS 'Schema referente ao sistema SISNET';
